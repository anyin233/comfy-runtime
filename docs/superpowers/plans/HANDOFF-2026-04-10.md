```python
"""Tests for the ported comfy_extras.nodes_model_advanced."""
import torch
from tests.fixtures.tiny_sd15 import make_tiny_sd15
from comfy_runtime.compat.comfy.model_patcher import ModelPatcher
from comfy_runtime.compat.comfy_extras import nodes_model_advanced as nma


def test_model_sampling_discrete_returns_clone():
    comp = make_tiny_sd15()
    model = ModelPatcher(comp["unet"])
    (out,) = nma.ModelSamplingDiscrete().patch(
        model=model, sampling="v_prediction", zsnr=False,
    )
    # Returns a clone (different object) but same underlying model
    assert out is not model
    assert out.model is model.model


def test_rescale_cfg_returns_modified_patcher():
    comp = make_tiny_sd15()
    model = ModelPatcher(comp["unet"])
    (out,) = nma.RescaleCFG().patch(model=model, multiplier=0.7)
    assert out is not model
    # The rescale function should be in model_options
    assert "sampler_cfg_function" in out.model_options or \
           "post_cfg_function" in out.model_options


def test_rescale_cfg_function_preserves_std():
    """The post-CFG rescale should reduce the effective CFG strength
    by ``multiplier`` while preserving the per-channel std of the
    pre-CFG conditional prediction."""
    cond = torch.randn(1, 4, 8, 8)
    uncond = torch.randn(1, 4, 8, 8)
    cfg = 8.0
    multiplier = 0.7

    # Standard CFG
    cfg_pred = uncond + cfg * (cond - uncond)
    # Rescaled
    rescaled = nma._rescale_cfg(cfg_pred, cond, multiplier)
    # The output should be a blend of cfg_pred and a normalized version
    assert rescaled.shape == cfg_pred.shape
    assert not torch.allclose(rescaled, cfg_pred)
```

**Suggested implementation sketch:**

```python
import torch

from comfy_runtime.compat.comfy.model_patcher import ModelPatcher


def _rescale_cfg(cfg_pred: torch.Tensor, cond_pred: torch.Tensor,
                 multiplier: float) -> torch.Tensor:
    """Rescale the post-CFG prediction to preserve cond's std.

    Reference: 'Common Diffusion Noise Schedules and Sample Steps Are
    Flawed' (Lin et al., 2024).
    """
    # Per-channel std along all non-batch dims
    spatial = list(range(1, cfg_pred.dim()))
    std_cond = cond_pred.std(dim=spatial, keepdim=True)
    std_cfg = cfg_pred.std(dim=spatial, keepdim=True)
    rescaled = cfg_pred * (std_cond / std_cfg.clamp(min=1e-7))
    return multiplier * rescaled + (1.0 - multiplier) * cfg_pred


class ModelSamplingDiscrete:
    """Switch a model between EPS / V_PREDICTION / LCM / X0 sampling
    parameterizations."""

    SAMPLING_TYPES = ["eps", "v_prediction", "lcm", "x0"]

    @classmethod
    def INPUT_TYPES(s):
        return {
            "required": {
                "model": ("MODEL",),
                "sampling": (s.SAMPLING_TYPES,),
                "zsnr": ("BOOLEAN", {"default": False}),
            }
        }

    RETURN_TYPES = ("MODEL",)
    FUNCTION = "patch"
    CATEGORY = "advanced/model"

    def patch(self, model, sampling, zsnr):
        out = model.clone()
        out.model_options.setdefault("transformer_options", {})
        out.model_options["transformer_options"]["model_sampling_type"] = sampling
        out.model_options["transformer_options"]["zsnr"] = bool(zsnr)
        return (out,)


class RescaleCFG:
    """Apply terminal-SNR-safe CFG rescaling via a post-cfg hook."""

    @classmethod
    def INPUT_TYPES(s):
        return {
            "required": {
                "model": ("MODEL",),
                "multiplier": ("FLOAT", {"default": 0.7, "min": 0.0,
                                          "max": 1.0, "step": 0.01}),
            }
        }

    RETURN_TYPES = ("MODEL",)
    FUNCTION = "patch"
    CATEGORY = "advanced/model"

    def patch(self, model, multiplier):
        out = model.clone()

        def _post_cfg(args):
            cond_pred = args["cond"]
            cfg_pred = args["cfg_result"]
            return _rescale_cfg(cfg_pred, cond_pred, multiplier)

        # Use whichever hook keys ModelPatcher actually consumes —
        # set_model_sampler_post_cfg_function exists in our compat
        if hasattr(out, "set_model_sampler_post_cfg_function"):
            out.set_model_sampler_post_cfg_function(_post_cfg)
        else:
            out.model_options["sampler_post_cfg_function"] = _post_cfg

        return (out,)

def rescale_zero_terminal_snr_sigmas(sigmas: torch.Tensor) -> torch.Tensor:
    """ZSNR sigma rescaling — used by AnimateDiff and other ZSNR fans."""
    alpha_cumprod = 1.0 / (sigmas ** 2 + 1.0)
    alpha = torch.sqrt(alpha_cumprod)
    sqrt_alpha_T = alpha[-1]
    sqrt_alpha_0 = alpha[0]
    alpha = (alpha - sqrt_alpha_T) / (sqrt_alpha_0 - sqrt_alpha_T)
    alpha = (alpha * sqrt_alpha_0).clamp(min=1e-10)
    new_sigmas = torch.sqrt(1.0 - alpha ** 2) / alpha
    return new_sigmas
```

## Background

The project goal: rewrite the GPL-licensed `_vendor/` ComfyUI inference
code into MIT-licensed `compat/` modules so the wheel ships pure MIT
without any GPL transitive code.  Optimization parity (multi-GPU,
lowvram, weight hot-swap) is non-negotiable per the user's requirement.

After Phases 0-4, `_vendor_bridge.py` was deleted entirely and the
wheel is GPL-free.  Phase 5 fills in the remaining functionality
(Flux sampling, ControlNet forward, custom sampler / mask / model
nodes) on top of the now-pure compat layer.

Benchmark comparison work (5.1, 5.2 in the original plan) is
explicitly **out of scope** for this branch — the user has indicated
the existing benchmark suite on `main` should be used for that.
